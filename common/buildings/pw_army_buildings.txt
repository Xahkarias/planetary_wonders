			trigger = {
				exists = owner
				owner = { has_tradition = tr_pw_monumentality_architecture_parlante }
			}
			unity = @pw_unity_production
		}
	}

		potential = {
			NOT = { has_modifier = pw_mod_integrated_monuments }
		}
		modifier = {
