# ENFORCER OR TELEPATH
triggered_planet_modifier = {
    potential = {
        exists = owner
        owner = { is_regular_empire = yes }
        owner = {
            NOR = {
                has_active_tradition = tr_psionics_psi_corps
                has_active_tradition = tr_psionics_shroud_psi_corps
            }
        }
    }
    modifier = {
        job_enforcer_add = $ENFORCER_AMOUNT$
    }
}
