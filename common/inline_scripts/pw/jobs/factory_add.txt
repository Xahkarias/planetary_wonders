triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			$OWNER_CONDITION$
			is_regular_empire = yes
		}
	}
	job_artisan_add = 100
	mult = $AMOUNT$
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = { 
			$OWNER_CONDITION$
			is_gestalt = yes
            is_wilderness_empire = no
		}
	}
	job_artisan_drone_add = 100
	mult = $AMOUNT$
}
